Пример выполнения BPMN модели

Этот пример показывает базовую обработку BPMN-процесса.

from bpmn_engine import ProcessEngine  

def execute_bpmn():  
    engine = ProcessEngine()  
    engine.load_model("example.bpmn")  
    engine.start_process("Process_1")  

execute_bpmn()
